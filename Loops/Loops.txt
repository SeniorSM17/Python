Python Program to Find the Sum of the Series 1/1!+1/2!+1/3!+…1/N!
# In[35]:


def sumOfSeries(num): 
    res = 0
    fact = 1
       
    for i in range(1, num+1): 
        fact *= i 
        res = res + (i/ fact) 
           
    return res 
       
n=int(input("Enter the number of terms: "))
 
print("Sum: ", sumOfSeries(n))
